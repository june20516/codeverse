## Dictionary

```swift
//생성
let fruitsColor: [String: String] = [
    "apple": "red"
    "banana": "yellow"
    "orange": "orange"
]

//호출
let colorOfApple = fruitsColor["apple"]
print(colorOfApple)   // red

//key만, value만 
let fruits = fruitsColor.keys    // ["apple","banana","orange"]
let colors = fruitsColor.value  //  ["orange","red","yellow"]

//아이템 삽입
fruitsColor["blueberry"] = "blue"   // "blueberry": "blue"가 들어감

//아이템 변경
fruitsColor["blueberry"] = "violet"  // "blueberry"의 value가 "violet"으로 바뀜
if let oldValue = fruitsColor.updateValue("violet", forKey: "blueberry") {
    print("The old value for blueberry was \(oldValue).")
}

//아이템 삭제
fruitsColor["orange"] = nil
//Return 있는 아이템 삭제
if let removedValue = fruitsColor.removeValue(forKey: "orange") {
    print("The removed airport's name is \(removedValue).")
} else {
    print("The fruitsColor dictionary does not contain a value for orange.")
}

// 기타 프로퍼티
let howMannyFruits = frueitsColor.count  //  개수 = 3
print(fruitsColor.isEmpty)    // 비어있는지 true 

// for-in에서 쓰기
for (fruit, color) in fruitsColor {
    print("\(fruit) is \(color)")
}

```
